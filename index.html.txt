<script>

/* ===== GRUPOS ===== */
const grupos = {
A:["Pilotis","México","Coreia do Sul","Suíça"],
B:["Brasil","Marrocos","Escócia","Canadá"],
C:["Holanda","Japão","Costa do Marfim","Curaçao"],
D:["Espanha","Alemanha","Uruguai","Austrália"],
E:["França","Senegal","Noruega","Equador"],
F:["Argentina","Bélgica","Argélia","Bósnia"],
G:["Portugal","Colômbia","Egito","Uzbequistão"],
H:["Inglaterra","Croácia","Arábia Saudita","Nova Zelândia"]
};

let jogos=[];

/* ===== ABAS ===== */
function abrir(tela){
    if(tela==="grupos") gruposFunc();
    if(tela==="tabela") tabelaFunc();
    if(tela==="mata") mataFunc();
}

/* ===== GRUPOS ===== */
function gruposFunc(){

jogos=[];

let html="<h2>⚽ FASE DE GRUPOS (13/06 - 28/06)</h2>";

for(let g in grupos){

let t=grupos[g];

html+=`<div class="group"><h3>Grupo ${g}</h3>`;

let ordem=[[0,1],[2,3],[0,2],[1,3],[0,3],[1,2]];

ordem.forEach((p,i)=>{

jogos.push({t1:t[p[0]],t2:t[p[1]]});

html+=`
<div class="match">
${t[p[0]]} <input type="number"> x <input type="number"> ${t[p[1]]}
</div>
`;

});

html+="</div>";
}

html+=`<button onclick="calcular()">GERAR TABELA / MATA-MATA</button>`;

document.getElementById("conteudo").innerHTML=html;
}

/* ===== TABELA + MATA ===== */
function calcular(){

let pts={};

jogos.forEach((j,i)=>{

let a=document.querySelectorAll("input")[i*2].value;
let b=document.querySelectorAll("input")[i*2+1].value;

a=+a;b=+b;

if(!pts[j.t1]) pts[j.t1]=0;
if(!pts[j.t2]) pts[j.t2]=0;

if(a>b) pts[j.t1]+=3;
else if(b>a) pts[j.t2]+=3;
else{pts[j.t1]++;pts[j.t2]++;}

});

window.resultado = Object.keys(pts).sort((a,b)=>pts[b]-pts[a]);

tabelaFunc();
}

/* ===== TABELA ===== */
function tabelaFunc(){

let html="<h2>📊 CLASSIFICAÇÃO ESTILO GE</h2>";

if(!window.resultado){
    html+="<p>Preencha os jogos primeiro.</p>";
    document.getElementById("conteudo").innerHTML=html;
    return;
}

window.resultado.forEach((t,i)=>{

let cor=i<2?"classified":"eliminated";

html+=`
<div class="line ${cor}">
<span>${i+1} ${t}</span>
<span>${Math.floor(Math.random()*10)} pts</span>
</div>
`;

});

document.getElementById("conteudo").innerHTML=html;
}

/* ===== MATA ===== */
function mataFunc(){

let r=window.resultado || [];

if(r.length<16){
document.getElementById("conteudo").innerHTML="<p>Complete os grupos primeiro.</p>";
return;
}

let html="<h2>🏆 MATA-MATA</h2>";

for(let i=0;i<16;i+=2){
html+=`<div class="match">${r[i]} 🆚 ${r[i+1]}</div>`;
}

document.getElementById("conteudo").innerHTML=html;
}
